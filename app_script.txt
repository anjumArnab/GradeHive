// Google Apps Script to handle CRUD operations with Google Sheets

// Global variables
const SHEET_NAME = "Students";
const SHEET_HEADERS = ["ID", "Name", "Result"];

/**
 * Setup the Google Sheet with necessary headers
 */
function setupSheet() {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = spreadsheet.getSheetByName(SHEET_NAME);
  
  // If sheet doesn't exist, create it
  if (!sheet) {
    sheet = spreadsheet.insertSheet(SHEET_NAME);
    sheet.appendRow(SHEET_HEADERS);
    sheet.getRange(1, 1, 1, SHEET_HEADERS.length).setFontWeight("bold");
  }
}

/**
 * Web app initialization
 */
function doGet() {
  return ContentService.createTextOutput("Student Management API is running.")
    .setMimeType(ContentService.MimeType.TEXT);
}

/**
 * Handle POST requests for CRUD operations
 */
function doPost(e) {
  const params = JSON.parse(e.postData.contents);
  const action = params.action;
  
  switch (action) {
    case "create":
      return createStudent(params.student);
    case "read":
      return getAllStudents();
    case "get":
      return getStudent(params.id);
    case "update":
      return updateStudent(params.student);
    case "delete":
      return deleteStudent(params.id);
    default:
      return ContentService.createTextOutput(JSON.stringify({
        status: "error",
        message: "Invalid action"
      })).setMimeType(ContentService.MimeType.JSON);
  }
}

/**
 * Create a new student record
 */
function createStudent(student) {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = spreadsheet.getSheetByName(SHEET_NAME);
  
  // Validate student data
  if (!student.id || !student.name) {
    return ContentService.createTextOutput(JSON.stringify({
      status: "error",
      message: "Student ID and name are required"
    })).setMimeType(ContentService.MimeType.JSON);
  }
  
  // Check if ID already exists
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === student.id) {
      return ContentService.createTextOutput(JSON.stringify({
        status: "error",
        message: "Student ID already exists"
      })).setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  // Add new student
  sheet.appendRow([student.id, student.name, student.result || ""]);
  
  return ContentService.createTextOutput(JSON.stringify({
    status: "success",
    message: "Student added successfully",
    student: student
  })).setMimeType(ContentService.MimeType.JSON);
}

/**
 * Get all students
 */
function getAllStudents() {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = spreadsheet.getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  
  const students = [];
  for (let i = 1; i < data.length; i++) {
    students.push({
      id: data[i][0],
      name: data[i][1],
      result: data[i][2]
    });
  }
  
  return ContentService.createTextOutput(JSON.stringify({
    status: "success",
    students: students
  })).setMimeType(ContentService.MimeType.JSON);
}

/**
 * Get a specific student by ID
 */
function getStudent(id) {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = spreadsheet.getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === id) {
      const student = {
        id: data[i][0],
        name: data[i][1],
        result: data[i][2]
      };
      
      return ContentService.createTextOutput(JSON.stringify({
        status: "success",
        student: student
      })).setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  return ContentService.createTextOutput(JSON.stringify({
    status: "error",
    message: "Student not found"
  })).setMimeType(ContentService.MimeType.JSON);
}

/**
 * Update an existing student
 */
function updateStudent(student) {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = spreadsheet.getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === student.id) {
      sheet.getRange(i + 1, 2).setValue(student.name);
      sheet.getRange(i + 1, 3).setValue(student.result || "");
      
      return ContentService.createTextOutput(JSON.stringify({
        status: "success",
        message: "Student updated successfully",
        student: student
      })).setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  return ContentService.createTextOutput(JSON.stringify({
    status: "error",
    message: "Student not found"
  })).setMimeType(ContentService.MimeType.JSON);
}

/**
 * Delete a student by ID
 */
function deleteStudent(id) {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = spreadsheet.getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === id) {
      sheet.deleteRow(i + 1);
      
      return ContentService.createTextOutput(JSON.stringify({
        status: "success",
        message: "Student deleted successfully"
      })).setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  return ContentService.createTextOutput(JSON.stringify({
    status: "error",
    message: "Student not found"
  })).setMimeType(ContentService.MimeType.JSON);
}

 * Set up the deployment as web app
 * This function is for manual execution in the script editor
 */
function setUp() {
