// Global variables
const SHEET_NAME = "Students";
const SHEET_HEADERS = ["ID", "Name", "Result"];

/**
 * Setup the Google Sheet with necessary headers
 */
function setupSheet() {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = spreadsheet.getSheetByName(SHEET_NAME);
  
  // If sheet doesn't exist, create it
  if (!sheet) {
    sheet = spreadsheet.insertSheet(SHEET_NAME);
    sheet.appendRow(SHEET_HEADERS);
    sheet.getRange(1, 1, 1, SHEET_HEADERS.length).setFontWeight("bold");
  }
}

/**
 * Web app initialization - handle GET requests
 */
function doGet() {
  return ContentService.createTextOutput(JSON.stringify({
    status: "success",
    message: "Student Management API is running."
  }))
  .setMimeType(ContentService.MimeType.JSON);
}

/**
 * Handle OPTIONS requests for CORS preflight
 */
function doOptions(e) {
  var output = ContentService.createTextOutput();
  output.setMimeType(ContentService.MimeType.TEXT);
  
  var headers = e.parameter.headers;
  if (headers['Origin']) {
    // Set necessary CORS headers
    output.setContent('');
  }
  
  return output;
}

/**
 * Handle POST requests for CRUD operations
 */
function doPost(e) {
  const params = JSON.parse(e.postData.contents);
  const action = params.action;
  let result;
  
  switch (action) {
    case "create":
      result = createStudent(params.student);
      break;
    case "read":
      result = getAllStudents();
      break;
    case "get":
      result = getStudent(params.id);
      break;
    case "update":
      result = updateStudent(params.student);
      break;
    case "delete":
      result = deleteStudent(params.id);
      break;
    default:
      result = {
        status: "error",
        message: "Invalid action"
      };
  }
  
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

/**
 * Create a new student record
 */
function createStudent(student) {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = spreadsheet.getSheetByName(SHEET_NAME);
  
  // Validate student data
  if (!student.id || !student.name) {
    return {
      status: "error",
      message: "Student ID and name are required"
    };
  }
  
  // Check if ID already exists
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === student.id) {
      return {
        status: "error",
        message: "Student ID already exists"
      };
    }
  }
  
  // Add new student
  sheet.appendRow([student.id, student.name, student.result || ""]);
  
  return {
    status: "success",
    message: "Student added successfully",
    student: student
  };
}

/**
 * Get all students
 */
function getAllStudents() {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = spreadsheet.getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  
  const students = [];
  for (let i = 1; i < data.length; i++) {
    students.push({
      id: data[i][0],
      name: data[i][1],
      result: data[i][2]
    });
  }
  
  return {
    status: "success",
    students: students
  };
}

/**
 * Get a specific student by ID
 */
function getStudent(id) {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = spreadsheet.getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === id) {
      const student = {
        id: data[i][0],
        name: data[i][1],
        result: data[i][2]
      };
      
      return {
        status: "success",
        student: student
      };
    }
  }
  
  return {
    status: "error",
    message: "Student not found"
  };
}

/**
 * Update an existing student
 */
function updateStudent(student) {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = spreadsheet.getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === student.id) {
      sheet.getRange(i + 1, 2).setValue(student.name);
      sheet.getRange(i + 1, 3).setValue(student.result || "");
      
      return {
        status: "success",
        message: "Student updated successfully",
        student: student
      };
    }
  }
  
  return {
    status: "error",
    message: "Student not found"
  };
}

/**
 * Delete a student by ID
 */
function deleteStudent(id) {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = spreadsheet.getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === id) {
      sheet.deleteRow(i + 1);
      
      return {
        status: "success",
        message: "Student deleted successfully"
      };
    }
  }
  
  return {
    status: "error",
    message: "Student not found"
  };
}
